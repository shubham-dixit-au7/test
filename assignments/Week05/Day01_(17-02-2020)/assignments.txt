								def closestNumbers(arr):
								    arr.sort()
								    arr1 = []
								    diff = 1000000000000001
								    for i in range(1, len(arr)):
								        d = abs(arr[i]-arr[i-1])
								        if d < diff:
								            arr1 = [arr[i-1], arr[i]]
								            diff = d
								        elif d == diff:
								            arr1.extend([arr[i-1], arr[i]])

								    return arr1
